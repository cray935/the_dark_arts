# Challenges are random chance values based on the difficulty of the task and the character's expert level.

ctda_necromancy_challenge_novice_success_chance = {
	value = 0

	if = {
		limit = {
			has_trait = ctda_necromancer_1
		}
		add = 60
	}
	else_if = {
		limit = {
			has_trait = ctda_necromancer_2
		}
		add = 70
	}
	else_if = {
		limit = {
			has_trait = ctda_necromancer_3
		}
		add = 80
	}
	else_if = {
		limit = {
			has_trait = ctda_necromancer_4
		}
		add = 90
	}
	else_if = {
		limit = {
			has_trait = ctda_necromancer_5
		}
		add = 100
	}

	if = {
		limit = {
			learning >= very_high_skill_rating
		}
		add = 5
	}
	if = {
		limit = {
			intrigue >= very_high_skill_rating
		}
		add = 5
	}
}
